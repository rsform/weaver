Core Principles

  1. Clean and consistent syntax that prioritizes readability
  2. Pipeline-oriented flow where appropriate
  3. Explicit over implicit
  4. Thread-local and immutable by default
  5. Square brackets for generics/templates
  6. Minimal noise and ceremony
  7. Predictable formatting rules

Basic Syntax Elements

Keywords and Names

Functions

// Function declaration
fn add(x: i32, y: i32) -> i32 {
	x + y
}

// Async function
fn fetch_data() -> Result[Data, Error] with Async {
	// ...
}

// Generic function
fn map[T, U](value: T, f: fn(T) -> U) -> U {
	f(value)
}

Variables


// Immutable by default
let x = 5

// Mutable variables must be declared with 'let mut'
let mut count = 0

// Explicit type annotation
let name: String = "Alice"

// Pattern binding
let Point(x, y) = point

Types and Traits

// Type declaration
type Point {
	x: f64
	y: f64
}

// Enum (sum type)
type Option[T] {
	Some(T)
	None
}

// Trait declaration
trait Display {
	fn display(self) -> String
}

// Generic trait with associated type
trait Container[T] {
	type Item
	fn get(self) -> Option[T]
	fn insert(self, item: T)
}

Control Flow

Pattern Matching

// Basic match expression using pipeline
let description = value
	|> match {
		Ok(n) if n > 0 -> "Positive"
		Ok(n) if n < 0 -> "Negative"
		Ok(0) -> "Zero"
		Err(e) -> "Error: {e}"
	}

// Match with complex patterns
let command = input
	|> match {
		Command.Quit -> exit(0)
		Command.Move(x, y) -> move_cursor(x, y)
		Command.Text(str) -> insert_text(str)
		_ -> beep()
	}

Conditionals

// If expression
let status = if count > 0 {
	"Active"
} else {
	"Inactive"
}

// Guard statement for early returns
fn process(data: Option[Data]) -> Result[Output, Error] {
	guard let unwrapped = data else {
		return Err(Error.InvalidData)
	}
	process_unwrapped(unwrapped)
}

Loops

// For-in loop
for item in items {
	process(item)
}

// While loop
while condition {
	// ...
}

// Loop with iterator methods
items
	|> filter(pred)
	|> map(transform)
	|> collect()

Error Handling

Results and Options

// Using Result type
fn divide(x: f64, y: f64) -> Result[f64, DivisionError] {
	if y == 0.0 {
		Err(DivisionError.DivideByZero)
	} else {
		Ok(x / y)
	}
}

// Chaining Results
let result = value
	|> first_operation()
	|> and_then(second_operation)
	|> map(third_operation)
	|> map_err(CustomError)

// Pattern matching on Result
let output = result
	|> match {
		Ok(value) -> process(value)
		Err(e) -> handle_error(e)
	}

Effects and Side Effects

Effect Declaration and Usage

// Declaring effects
effect IO {
	read_line() -> String
	print(str: String)
}

// Using effects
fn greet() -> String with IO {
	print("What's your name?")
	let name = read_line()
	print("Hello, {name}!")
	name
}

// Handling effects
fn main() {
	run with IO = ConsoleIO {
		let name = greet()
		print("Goodbye, {name}")
	}
}

Generics and Traits

Generic Types

	head: Option[T]

}

// Generic trait bounds
fn sort[T: Ord](list: &mut [T]) {
	// Implementation
}

// Multiple bounds
fn process[T: Display + Clone](item: T) {
	// Implementation
}

Trait Implementation

implement Display for Point {
	fn display(self) -> String {
		"Point({self.x}, {self.y})"
	}
}

// Generic implementation
implement[T: Display] Display for Option[T] {
	fn display(self) -> String {
		match self {
			Some(value) -> "Some({value})"
			None -> "None"
		}
	}
}

Concurrency

Async/Await

// Async function declaration
fn fetch_data(url: String) -> Result[Data, Error] with Async {
	let response = http.get(url)?.await
	let data = response.json()?.await
	Ok(data)
}

// Using async functions
fn process() with Async {
	let data = fetch_data("example.com")?
		|> await
	process_data(data)
		|> await
}

Thread Safety

// Thread-local by default
let counter = 0

// Explicitly shared state
shared let global_counter = AtomicInteger(0)

// Shared function
shared fn increment() {
	global_counter.fetch_add(1)
}

Pipeline Operator

Basic Usage

// Pipeline with simple transformations
let result = value
	|> transform1()
	|> transform2()
	|> transform3()
	
// Pipeline with closures
let processed = data
	|> map(x -> x * 2)
	|> filter(x -> x > 0)
	|> fold(0, (acc, x) -> acc + x)

Complex Pipelines

let result = initial_value
	|> validate()
	|> match {
		Ok(value) -> process(value)
		Err(e) -> handle_error(e)
	}
	|> transform()
	|> finalize()

Formatting Rules

  1. Use 4 spaces for indentation
  2. One statement per line
  3. Blank line between function definitions
  4. No semicolon for expressions that are the last in a block
  5. Space after keywords
  6. No space after function names in calls
  7. Align match arms
  8. Pipeline operators at the start of lines
  9. Break long lines at sensible points (pipelines, binary operators)

Best Practices

  1. Prefer pipeline style for transformations and matches
  2. Use pattern matching over if/else chains
  3. Make effects explicit
  4. Keep functions focused and small
  5. Use meaningful and descriptive names
  6. Document public interfaces
  7. Utilize type inference where it aids readability
  8. Make thread sharing explicit
  9. Handle all error cases explicitly
  10. Use appropriate abstraction levels
